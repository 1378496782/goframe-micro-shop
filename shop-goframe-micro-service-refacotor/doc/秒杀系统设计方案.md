# 秒杀系统设计方案（Sentinel熔断限流＋令牌桶削峰）

## 1. 系统架构设计

### 1.1 整体架构图

```
┌─────────────┐     ┌──────────────┐     ┌────────────────┐
│  客户端请求  │ ──> │  Gateway-H5  │ ──> │  Sentinel网关限流  │
└─────────────┘     └──────────────┘     └────────────────┘
                                               │
                                               ▼
┌─────────────┐     ┌──────────────┐     ┌────────────────┐     ┌──────────────┐
│ 数据库存储   │ <── │  秒杀服务     │ <── │  令牌桶削峰组件  │ <── │  RabbitMQ    │
└─────────────┘     └──────┬───────┘     └────────────────┘     └──────────────┘
                           │                                        ▲
                           ▼                                        │
                     ┌──────────────┐                      ┌─────────────────┐
                     │  Redis缓存    │ <─────────────────── │  库存服务        │
                     └──────────────┘                      └─────────────────┘
```

### 1.2 核心组件说明

1. **Gateway-H5**：前端入口，接收秒杀请求
2. **Sentinel网关限流**：流量控制和熔断降级，保护后端服务
3. **令牌桶削峰组件**：平滑突发流量，避免系统过载
4. **RabbitMQ**：消息队列，用于异步处理秒杀请求和结果通知
5. **秒杀服务**：专门处理秒杀业务逻辑
6. **Redis缓存**：存储秒杀商品信息、库存和分布式锁
7. **库存服务**：处理库存扣减，基于现有的库存管理实现扩展
8. **数据库存储**：持久化秒杀相关数据

### 1.3 架构设计原则

1. **高可用**：通过熔断、降级、限流等机制确保系统在高并发下的可用性
2. **可扩展性**：模块化设计，方便横向扩展
3. **高性能**：使用缓存、异步处理等方式提高系统性能
4. **安全性**：防刷、防超卖等安全机制
5. **兼容性**：与现有系统无缝集成，不影响现有功能

## 2. 秒杀系统核心流程

### 2.1 秒杀流程详细设计

#### 2.1.1 预热流程

1. **定时任务触发**：系统启动时或秒杀活动开始前通过定时任务触发预热
2. **商品数据加载**：从数据库加载即将开始的秒杀商品信息
3. **库存初始化**：在Redis中初始化秒杀商品库存
4. **热点数据缓存**：缓存秒杀商品详情、活动规则等热点数据
5. **预热结果记录**：记录预热结果，用于监控和告警

#### 2.1.2 请求处理流程

1. **请求接收**：Gateway-H5接收用户秒杀请求
2. **参数校验**：验证请求参数的合法性
3. **Sentinel限流**：经过Sentinel网关限流处理
4. **令牌桶削峰**：通过令牌桶算法平滑流量
5. **用户验证**：验证用户身份、登录状态、购买权限
6. **资格检查**：检查用户是否已购买过该秒杀商品（防重复购买）
7. **库存检查**：检查商品库存是否充足
8. **库存扣减**：使用Redis Lua脚本原子性扣减库存
9. **消息入队**：将秒杀成功的请求信息发送到RabbitMQ队列
10. **结果返回**：返回初步的秒杀结果（异步处理需要后续查询最终结果）

#### 2.1.3 异步处理流程

1. **消息消费**：秒杀服务消费者从队列获取秒杀请求
2. **订单创建**：创建秒杀订单记录
3. **库存确认**：再次确认并持久化库存扣减
4. **事务处理**：使用数据库事务确保数据一致性
5. **结果记录**：记录秒杀结果到Redis和数据库
6. **消息通知**：发送订单创建成功通知

#### 2.1.4 结果查询流程

1. **查询请求**：用户查询秒杀结果
2. **缓存查询**：首先从Redis查询秒杀结果
3. **数据库查询**：Redis未命中时从数据库查询
4. **结果返回**：返回秒杀结果给用户

```go
// app/gateway-h5/internal/middleware/sentinel.go
package middleware

import (
    "context"
    "github.com/alibaba/sentinel-golang/api"
    "github.com/alibaba/sentinel-golang/core/base"
    "github.com/gogf/gf/v2/net/ghttp"
    "net/http"
    "time"
)

// SentinelMiddleware Sentinel中间件
func SentinelMiddleware() ghttp.HandlerFunc {
    return func(r *ghttp.Request) {
        // 获取请求路径作为资源名
        resourceName := r.RequestURI
        
        // 初始化上下文
        ctx := context.Background()
        
        // 执行限流控制
        entry, blockErr := api.Entry(
            resourceName,
            api.WithResourceType(base.ResTypeWeb),
            api.WithTrafficType(base.Inbound),
        )
        
        if blockErr != nil {
            // 被限流或熔断
            r.Response.WriteJsonExit(map[string]interface{}{
                "code":    429,
                "message": "当前请求人数过多，请稍后重试",
                "data":    nil,
            })
            return
        }
        
        // 确保在函数结束时退出entry
        defer entry.Exit()
        
        // 继续处理请求
        r.Middleware.Next()
    }
}
```

```go
// DoWithSentinel 使用Sentinel包装业务逻辑
func (s *SentinelService) DoWithSentinel(ctx context.Context, resourceName string, fn func(ctx context.Context) error) error {
    entry, blockErr := api.Entry(
        resourceName,
        api.WithResourceType(base.ResTypeCommon),
        api.WithTrafficType(base.Inbound),
    )
    
    if blockErr != nil {
        return gerror.New("当前请求人数过多，请稍后重试")
    }
    
    defer entry.Exit()
    
    return fn(ctx)
}
```

```go
func (s *FlashSaleServiceImpl) CreateFlashSaleOrder(ctx context.Context, req *CreateFlashSaleOrderReq) (*CreateFlashSaleOrderRes, error) {
    // 使用Sentinel包装业务逻辑
    var result *CreateFlashSaleOrderRes
    var err error
    
    sentinelErr := s.sentinelService.DoWithSentinel(ctx, "flash_sale_create_order", func(ctx context.Context) error {
        // 业务逻辑实现
        result, err = s.doCreateFlashSaleOrder(ctx, req)
        return err
    })
    
    // 处理限流或熔断情况
    if sentinelErr != nil {
        return &CreateFlashSaleOrderRes{
            Success:  false,
            Message:  "当前请求人数过多，请稍后重试",
            Status:   2,
        }, nil
    }
    
    return result, err
}
```

### 3.5 Sentinel规则调优建议

1. **预热期设置**：对于秒杀开始前的流量，可以设置预热期，让系统逐步提升处理能力
2. **排队等待策略**：对于秒杀请求，可以使用排队等待策略，而不是直接拒绝
3. **热点参数优先级**：对热门商品设置更低的QPS限制，防止单个商品被过度请求
4. **动态调整**：根据实际流量情况，动态调整限流参数
5. **监控告警阈值**：设置合理的监控告警阈值，及时发现问题

### 3.6 与令牌桶削峰的协同

Sentinel限流主要在网关层和服务层提供保护，而令牌桶削峰主要在业务层面提供更细粒度的流量控制。两者协同工作，可以：

1. Sentinel提供粗粒度的限流和熔断保护
2. 令牌桶提供更精细的流量平滑控制
3. 共同确保系统在秒杀场景下的稳定性和可用性

## 4. 令牌桶削峰设计

### 4.1 令牌桶实现原理

#### 4.1.1 基本原理

1. **令牌生成**：系统以固定的速率（rate）向桶中添加令牌
2. **令牌存储**：令牌桶有最大容量（burst）限制，多余的令牌会被丢弃
3. **请求处理**：请求到达时尝试从桶中获取令牌，获取成功才处理请求
4. **流量控制**：通过控制令牌生成速率和桶容量，实现流量的精确控制和平滑处理

#### 4.1.2 数学模型

- 令牌生成速率：r（个/秒）
- 桶最大容量：b（个令牌）
- 最后一次更新时间：t_last
- 当前令牌数：tokens
- 令牌补充公式：tokens = min(b, tokens + r * (t_now - t_last))

```go
// 令牌桶Lua脚本
const tokenBucketLuaScript = `
-- 令牌桶参数
local key = KEYS[1]
local rate = tonumber(ARGV[1])       -- 令牌生成速率（个/毫秒）
local burst = tonumber(ARGV[2])      -- 桶容量
local tokens = tonumber(ARGV[3])     -- 请求令牌数
local now = tonumber(ARGV[4])        -- 当前时间戳（毫秒）

-- 令牌桶结构：{last_refill_time, available_tokens}
local bucket = redis.call('hmget', key, 'last_refill_time', 'available_tokens')
local lastRefillTime = tonumber(bucket[1] or now)
local availableTokens = tonumber(bucket[2] or burst)

-- 计算需要补充的令牌数
timeElapsed = now - lastRefillTime
newTokens = timeElapsed * rate

-- 更新可用令牌数（不超过桶容量）
availableTokens = math.min(burst, availableTokens + newTokens)

-- 检查是否有足够的令牌
if availableTokens >= tokens then
    -- 扣除令牌
    availableTokens = availableTokens - tokens
    -- 更新令牌桶状态
    redis.call('hmset', key, 'last_refill_time', now, 'available_tokens', availableTokens)
    -- 设置过期时间（24小时）
    redis.call('expire', key, 86400)
    return 1
else
    -- 令牌不足
    return 0
end
`
```

```go
// adjustRates 调整令牌生成速率
func (m *AdaptiveTokenBucketManager) adjustRates(ctx context.Context) {
    // 获取系统负载信息
    load, err := m.monitor.GetSystemLoad(ctx)
    if err != nil {
        g.Log().Error(ctx, "获取系统负载失败", err)
        return
    }
    
    m.mu.RLock()
    defer m.mu.RUnlock()
    
    for key, config := range m.configs {
        // 根据系统负载动态调整速率
        var newRate float64
        if load.CPUUsage > 0.8 {
            // 系统负载高，降低速率
            newRate = math.Max(config.MinRate, config.BaseRate * 0.9)
        } else if load.CPUUsage < 0.5 {
            // 系统负载低，提高速率
            newRate = math.Min(config.MaxRate, config.BaseRate * 1.1)
        } else {
            // 系统负载正常，保持基础速率
            newRate = config.BaseRate
        }
        
        // 更新配置
        config.BaseRate = newRate
        
        // 将新的配置保存到Redis
        err := m.saveConfigToRedis(ctx, key, config)
        if err != nil {
            g.Log().Error(ctx, "保存令牌桶配置到Redis失败", err)
        }
    }
}
```

```go
type WarmupTokenBucketService struct {
    tokenBucketService     *TokenBucketService
    adaptiveTokenManager   *AdaptiveTokenBucketManager
}

// NewWarmupTokenBucketService 创建预热期令牌桶服务
func NewWarmupTokenBucketService(tokenBucketService *TokenBucketService, adaptiveTokenManager *AdaptiveTokenBucketManager) *WarmupTokenBucketService {
    return &WarmupTokenBucketService{
        tokenBucketService:     CHANGE_ME_SECRET
        adaptiveTokenManager:   CHANGE_ME_SECRET
    }
}

// StartWarmup 开始预热
func (s *WarmupTokenBucketService) StartWarmup(ctx context.Context, activityId uint32, startTime time.Time, duration time.Duration) {
    // 预热开始时间
    now := time.Now()
    if now.After(startTime) {
        return // 活动已开始，不需要预热
    }
    
    // 计算预热间隔
    warmupDuration := startTime.Sub(now)
    if warmupDuration < time.Minute {
        warmupDuration = time.Minute // 最小预热时间1分钟
    }
    
    // 计算预热步骤
    steps := int(warmupDuration.Seconds() / 10) // 每10秒调整一次
    if steps < 1 {
        steps = 1
    }
    
    // 初始速率
    initialRate := 10.0 / 1000 // 10个/秒
    // 目标速率
    targetRate := 100.0 / 1000 // 100个/秒
    // 速率增量
    rateIncrement := (targetRate - initialRate) / float64(steps)
    
    // 启动预热协程
    go func() {
        ticker := time.NewTicker(10 * time.Second)
        defer ticker.Stop()
        
        currentRate := initialRate
        for i := 0; i < steps; i++ {
            select {
            case <-ctx.Done():
                return
            case <-ticker.C:
                // 更新令牌桶速率
                activityKey := GetActivityTokenBucketKey(activityId)
                
                // 注册新的令牌桶配置
                s.adaptiveTokenManager.RegisterTokenBucket(
                    &TokenBucketConfig{
                        Key:        activityKey,
                        BaseRate:   currentRate,
                        MaxRate:    targetRate,
                        MinRate:    initialRate,
                        Burst:      int(currentRate * 2000), // 2秒的突发容量
                        AdjustStep: 0.01,
                    },
                )
                
                g.Log().Info(ctx, "预热进度", g.Map{
                    "activity_id": activityId,
                    "step":        i + 1,
                    "total_steps": steps,
                    "current_rate": currentRate * 1000, // 转换为秒
                })
                
                // 增加速率
                currentRate += rateIncrement
                if currentRate > targetRate {
                    currentRate = targetRate
                }
                
                // 检查是否已到活动开始时间
                if time.Now().After(startTime) {
                    break
                }
            }
        }
        
        // 预热结束，设置最终速率
        s.adaptiveTokenManager.RegisterTokenBucket(
            &TokenBucketConfig{
                Key:        GetActivityTokenBucketKey(activityId),
                BaseRate:   targetRate,
                MaxRate:    targetRate * 1.2, // 允许1.2倍的突发
                MinRate:    targetRate * 0.5, // 最小为目标速率的一半
                Burst:      int(targetRate * 2000),
                AdjustStep: 0.01,
            },
        )
        
        g.Log().Info(ctx, "预热完成", g.Map{
            "activity_id": activityId,
            "final_rate":  targetRate * 1000,
        })
    }()
}
```

```go
// NewCacheService 创建缓存服务实例
func NewCacheService(redisClient *redis.Client) *CacheService {
    return &CacheService{

// GetFlashSaleGoods 获取秒杀商品信息
func (s *CacheService) GetFlashSaleGoods(ctx context.Context, goodsId int64) (*model.FlashSaleGoods, error) {
    // 先查本地缓存
    key := fmt.Sprintf(consts.FlashSaleGoodsKey, goodsId)
    cached, err := s.localCache.Get(ctx, key)
    if err == nil && cached != nil {
        if goods, ok := cached.(*model.FlashSaleGoods); ok {
            return goods, nil
        }
    }
    
    // 查Redis缓存
    data, err := s.redisClient.Get(ctx, key).Result()
    if err != nil {
        return nil, gerror.Wrap(err, "获取秒杀商品缓存失败")
    }
    
    goods := &model.FlashSaleGoods{}
    err = gjson.DecodeTo(data, goods)
    if err != nil {
        return nil, gerror.Wrap(err, "解析秒杀商品缓存失败")
    }
    
    // 更新本地缓存
    s.localCache.Set(ctx, key, goods, time.Minute)
    
    return goods, nil
}

// SetFlashSaleGoods 设置秒杀商品信息
func (s *CacheService) SetFlashSaleGoods(ctx context.Context, goods *model.FlashSaleGoods) error {
    key := fmt.Sprintf(consts.FlashSaleGoodsKey, goods.GoodsId)
    data, err := gjson.Encode(goods)
    if err != nil {
        return gerror.Wrap(err, "序列化秒杀商品失败")
    }
    
    err = s.redisClient.Set(ctx, key, data, time.Duration(consts.FlashSaleGoodsCacheExpire)*time.Second).Err()
    if err != nil {
        return gerror.Wrap(err, "设置秒杀商品缓存失败")
    }
    
    // 更新本地缓存
    s.localCache.Set(ctx, key, goods, time.Minute)
    
    return nil
}

// GetFlashSaleStock 获取秒杀库存

    key := fmt.Sprintf(consts.FlashSaleStockKey, goodsId)
    stockStr, err := s.redisClient.Get(ctx, key).Result()
    if err != nil {
        return 0, gerror.Wrap(err, "获取秒杀库存失败")
```

```go
// 库存扣减Lua脚本
const stockDecreaseScript = `
local stockKey = KEYS[1]
local userLimitKey = KEYS[2]
local maxBuy = tonumber(ARGV[1])
local buyCount = tonumber(ARGV[2])

-- 检查用户购买限制

else
    redis.call('set', userLimitKey, buyCount, 'EX', 86400)

return currentStock - buyCount -- 返回剩余库存
`

// DecreaseStock 扣减库存
func (s *StockService) DecreaseStock(ctx context.Context, goodsId, userId int64, buyCount int) (bool, error) {
    // 获取商品信息
    goods, err := s.cacheService.GetFlashSaleGoods(ctx, goodsId)
    if err != nil {
        return false, gerror.Wrap(err, "获取商品信息失败")
    }
    
    // 构建键
    stockKey := fmt.Sprintf(consts.FlashSaleStockKey, goodsId)
    userLimitKey := fmt.Sprintf(consts.FlashSaleUserLimitKey, userId, goodsId)
    
    // 执行Lua脚本
    result, err := s.redisClient.Eval(
        ctx,
        stockDecreaseScript,
        []string{stockKey, userLimitKey},
        goods.MaxBuyPerUser,
        buyCount,
    ).Result()
    
    if err != nil {
        return false, gerror.Wrap(err, "执行库存扣减脚本失败")
    }
    
    // 解析结果
    remainingStock := gconv.Int(result)
    if remainingStock < 0 {
        return false, gerror.New("超过购买限制")
    } else if remainingStock == 0 {
        return false, gerror.New("库存不足")
    }
    
    return true, nil
}
```

```go
// WarmupFlashSaleGoods 预热秒杀商品缓存
func (s *WarmupService) WarmupFlashSaleGoods(ctx context.Context) error {
    // 获取即将开始和进行中的活动
    activities, err := dao.FlashSaleActivity.Find(ctx, g.Map{
        "status": []int{0, 1}, // 未开始和进行中
    })
    if err != nil {
        return gerror.Wrap(err, "获取活动列表失败")

            "activity_id": activity.Id,
            "status":      1, // 已上架
        })

            // 预热商品信息
            err = s.cacheService.SetFlashSaleGoods(ctx, &goods)
            if err != nil {
                g.Log().Error(ctx, "预热商品信息失败", g.Map{
                    "goods_id": goods.GoodsId,
                    "error":    err,
                })
                continue
            }
            
            // 预热库存
            err = s.cacheService.SetFlashSaleStock(ctx, goods.GoodsId, goods.Stock)
            if err != nil {
                g.Log().Error(ctx, "预热商品库存失败", g.Map{
                    "goods_id": goods.GoodsId,
                    "error":    err,
                })
            }
        }
    }
    
    return nil
}
```

```go
        cacheService:   cacheService,
        messageService: messageService,
    }

// SyncStockToCache 将数据库库存同步到缓存
func (s *StockSyncService) SyncStockToCache(ctx context.Context) error {
    // 获取所有秒杀商品
    goodsList, err := dao.FlashSaleGoods.Find(ctx, g.Map{
        "status": 1, // 已上架
    })
    if err != nil {
        return gerror.Wrap(err, "获取秒杀商品失败")
    }
    
    for _, goods := range goodsList {
        // 比较数据库和缓存中的库存
        cacheStock, err := s.cacheService.GetFlashSaleStock(ctx, goods.GoodsId)
        if err != nil {
            // 缓存不存在，直接设置
            err = s.cacheService.SetFlashSaleStock(ctx, goods.GoodsId, goods.Stock)
            if err != nil {
                g.Log().Error(ctx, "同步库存到缓存失败", g.Map{
                    "goods_id": goods.GoodsId,
                    "error":    err,
                })
            }
            continue
        }
        
        // 如果缓存库存和数据库不一致，以数据库为准
        if cacheStock != goods.Stock {
            err = s.cacheService.SetFlashSaleStock(ctx, goods.GoodsId, goods.Stock)
            if err != nil {
                g.Log().Error(ctx, "修复库存不一致失败", g.Map{
                    "goods_id": goods.GoodsId,
                    "db_stock": goods.Stock,
                    "cache_stock": cacheStock,
                    "error": err,
                })
            } else {
                g.Log().Info(ctx, "修复库存不一致", g.Map{
                    "goods_id": goods.GoodsId,
                    "old_stock": cacheStock,
                    "new_stock": goods.Stock,
                })
            }
        }
    }
    
    return nil
}

// ScheduleSync 定时同步
func (s *StockSyncService) ScheduleSync(ctx context.Context) {
    ticker := time.NewTicker(2 * time.Minute) // 每2分钟同步一次
    defer ticker.Stop()
    
    // 立即执行一次同步
    go s.SyncStockToCache(ctx)
    
    for {
        select {
        case <-ctx.Done():
```

```go
package service
```

```go
// RecoveryService 恢复服务
type RecoveryService struct {
    redisClient    *redis.Client
    stockService   *StockService
    messageService *MessageService
}

// NewRecoveryService 创建恢复服务实例
func NewRecoveryService(redisClient *redis.Client, stockService *StockService, messageService *MessageService) *RecoveryService {
    return &RecoveryService{
        redisClient:    redisClient,
        stockService:   stockService,
        messageService: messageService,
    }
}

// RecoverExpiredOrders 恢复过期未支付订单的库存
func (s *RecoveryService) RecoverExpiredOrders(ctx context.Context) error {
    // 获取15分钟前创建且未支付的订单
    expiredTime := time.Now().Add(-15 * time.Minute)
    expiredOrders, err := dao.FlashSaleOrder.Find(ctx, g.Map{
        "status":     0, // 待支付
        "created_at": g.Map{"<": expiredTime},
    })
    if err != nil {
        return gerror.Wrap(err, "获取过期订单失败")
    }
    
    for _, order := range expiredOrders {
        // 开启事务
        tx, err := g.DB().Begin(ctx)
        if err != nil {
            g.Log().Error(ctx, "开启事务失败", g.Map{
                "order_no": order.OrderNo,
                "error":    err,
            })
            continue
        }
        
        // 更新订单状态为已取消
        _, err = dao.FlashSaleOrder.Ctx(ctx).TX(tx).Where(g.Map{
            "id":     order.Id,
            "status": 0, // 确保还是待支付状态
        }).Update(g.Map{
            "status": 2, // 已取消
        })
        if err != nil {
            tx.Rollback()
            g.Log().Error(ctx, "更新订单状态失败", g.Map{
                "order_no": order.OrderNo,
                "error":    err,
            })
            continue
        }
        
        // 恢复库存
        _, err = dao.FlashSaleGoods.Ctx(ctx).TX(tx).Where(g.Map{
            "goods_id":    order.GoodsId,
            "activity_id": order.ActivityId,
        }).Increment("stock", order.GoodsCount)
        if err != nil {
            tx.Rollback()
            g.Log().Error(ctx, "恢复库存失败", g.Map{
                "order_no": order.OrderNo,
                "error":    err,
            })
            continue
        }
        
        // 提交事务
        err = tx.Commit()
        if err != nil {
            g.Log().Error(ctx, "提交事务失败", g.Map{
                "order_no": order.OrderNo,
                "error":    err,
            })
            continue
        }
        
        // 更新Redis缓存库存
        err = s.stockService.IncreaseStock(ctx, order.GoodsId, order.GoodsCount)
        if err != nil {
            g.Log().Error(ctx, "更新缓存库存失败", g.Map{
                "order_no": order.OrderNo,
                "error":    err,
            })
        }
        
        // 发布库存更新消息
        err = s.messageService.PublishStockUpdateMessage(ctx, order.GoodsId, order.GoodsCount)
        if err != nil {
            g.Log().Error(ctx, "发布库存更新消息失败", g.Map{
                "order_no": order.OrderNo,
                "error":    err,
            })
        }
        
        g.Log().Info(ctx, "恢复过期订单库存成功", g.Map{
            "order_no":    order.OrderNo,
            "goods_id":    order.GoodsId,
            "recover_stock": order.GoodsCount,
        })
    }
    
    return nil
}

// ScheduleRecovery 定时执行恢复任务
func (s *RecoveryService) ScheduleRecovery(ctx context.Context) {
    ticker := time.NewTicker(30 * time.Second) // 每30秒执行一次
    defer ticker.Stop()
    
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            go s.RecoverExpiredOrders(ctx)
        }
    }
}
```

```go
// Register 注册服务
func (s *RegistryService) Register(ctx context.Context) error {
    // 生成服务key
    serviceKey := fmt.Sprintf("/services/%s/%s", s.config.ServiceName, s.serviceID)
    
    // 生成服务地址
    serviceAddr := fmt.Sprintf("%s:%d", s.config.Host, s.config.Port)
    
    // 创建租约
    lease, err := s.etcdClient.Grant(ctx, s.config.TTL)
    if err != nil {
        return gerror.Wrap(err, "创建租约失败")
    }
    
    // 注册服务
    _, err = s.etcdClient.Put(ctx, serviceKey, serviceAddr, clientv3.WithLease(lease.ID))
    if err != nil {
        return gerror.Wrap(err, "注册服务失败")
    }
    
    // 启动续约
    keepAliveChan, err := s.etcdClient.KeepAlive(ctx, lease.ID)
    if err != nil {
        return gerror.Wrap(err, "启动续约失败")
    }
    
    // 处理续约
    go func() {
        for {
            select {
            case _, ok := <-keepAliveChan:
                if !ok {
                    s.logger.Warning(ctx, "续约通道关闭")
                    return
                }
                s.logger.Debug(ctx, "服务续约成功")
            case <-ctx.Done():
                s.logger.Info(ctx, "服务停止续约")
                return
            }
        }
    }()
    
    s.logger.Info(ctx, "服务注册成功", g.Map{
        "service_id":   s.serviceID,
        "service_name": s.config.ServiceName,
        "address":      serviceAddr,
    })
    
    return nil
}

        "service_id":   s.serviceID,

    })
    
    return nil
```

```go
package service

import (
    "context"
    "fmt"
    "time"
    
    "github.com/gogf/gf/v2/errors/gerror"
    "github.com/gogf/gf/v2/frame/g"

)

// CircuitBreakerConfig 熔断器配置
type CircuitBreakerConfig struct {
    Name          string
    MaxRequests   uint32
    Interval      time.Duration
    Timeout       time.Duration
    FailureThreshold float64
}

// CircuitBreakerService 熔断器服务
type CircuitBreakerService struct {
    breakers map[string]*gobreaker.CircuitBreaker
}

// NewCircuitBreakerService 创建熔断器服务实例
func NewCircuitBreakerService() *CircuitBreakerService {
    return &CircuitBreakerService{
        breakers: make(map[string]*gobreaker.CircuitBreaker),
    }
}

// GetCircuitBreaker 获取熔断器
func (s *CircuitBreakerService) GetCircuitBreaker(name string, config *CircuitBreakerConfig) *gobreaker.CircuitBreaker {
    if breaker, exists := s.breakers[name]; exists {
        return breaker
    }
    
    // 创建熔断器配置
    if config == nil {
        config = &CircuitBreakerConfig{
            Name:             name,
            MaxRequests:      10,
            Interval:         60 * time.Second,
            Timeout:          30 * time.Second,
            FailureThreshold: 0.5,
        }
    }
    
    settings := gobreaker.Settings{
        Name:        config.Name,
        MaxRequests: config.MaxRequests,
        Interval:    config.Interval,
        Timeout:     config.Timeout,
        ReadyToTrip: func(counts gobreaker.Counts) bool {
            failureRatio := float64(counts.TotalFailures) / float64(counts.Requests)
            return counts.Requests >= 10 && failureRatio >= config.FailureThreshold
        },
        OnStateChange: func(name string, from gobreaker.State, to gobreaker.State) {
            g.Log().Warning(context.Background(), "熔断器状态变更", g.Map{
                "name":       name,
                "from_state": from.String(),
                "to_state":   to.String(),
            })
        },
    }
    
    // 创建熔断器
    breaker := gobreaker.NewCircuitBreaker(settings)
    s.breakers[name] = breaker
    
    return breaker
}

// ExecuteWithCircuitBreaker 使用熔断器执行
func (s *CircuitBreakerService) ExecuteWithCircuitBreaker(ctx context.Context, name string, fallback func() (interface{}, error), fn func() (interface{}, error)) (interface{}, error) {
    breaker := s.GetCircuitBreaker(name, nil)
    
    result, err := breaker.Execute(func() (interface{}, error) {
        return fn()
    })
    
    if err != nil {
        // 如果熔断器开启，执行降级逻辑
        if gerror.Is(err, gobreaker.ErrOpenState) {
            g.Log().Warning(ctx, "熔断器开启，执行降级逻辑", g.Map{
                "name": name,
            })
            return fallback()
        }
        return nil, err
    }
    
    return result, nil
}
```

```go
    // 创建带超时的context
    timeoutCtx, cancel := context.WithTimeout(ctx, timeout)
    defer cancel()
    
    // 创建错误通道
    errChan := make(chan error, 1)
    
    // 异步执行
    go func() {
        errChan <- fn(timeoutCtx)
    }()
    
    // 等待结果或超时
    select {
    case err := <-errChan:
        return err
    case <-timeoutCtx.Done():
        return gerror.New("操作超时")
    }
}
```

```go
// DatabaseFailoverService 数据库故障转移服务
type DatabaseFailoverService struct {
    mu            sync.RWMutex
    failedSlaves  map[string]bool
    masterHealthy bool
}

// NewDatabaseFailoverService 创建数据库故障转移服务实例
func NewDatabaseFailoverService() *DatabaseFailoverService {
    return &DatabaseFailoverService{
        failedSlaves:  make(map[string]bool),
        masterHealthy: true,
    }
}

// StartHealthCheck 开始健康检查
func (s *DatabaseFailoverService) StartHealthCheck(ctx context.Context) {
    // 检查从库健康
    go s.checkSlaveHealth(ctx)
    
    // 检查主库健康
    go s.checkMasterHealth(ctx)
}

// checkSlaveHealth 检查从库健康
func (s *DatabaseFailoverService) checkSlaveHealth(ctx context.Context) {
    ticker := time.NewTicker(5 * time.Second)
    defer ticker.Stop()
    
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            // 检查从库1
            s.checkSlave(ctx, "slave1")
            
            // 检查从库2
            s.checkSlave(ctx, "slave2")
        }
    }
}

// checkSlave 检查单个从库
func (s *DatabaseFailoverService) checkSlave(ctx context.Context, slaveName string) {
    // 简单的健康检查查询
    _, err := g.DB(slaveName).GetAll(ctx, "SELECT 1")
    
    s.mu.Lock()
    defer s.mu.Unlock()
    
    if err != nil {
        // 从库不可用
        if !s.failedSlaves[slaveName] {
            g.Log().Warning(ctx, "从库不可用", g.Map{
                "slave": slaveName,
                "error": err.Error(),
            })
            s.failedSlaves[slaveName] = true
        }
    } else {
        // 从库恢复
        if s.failedSlaves[slaveName] {
            g.Log().Info(ctx, "从库恢复", g.Map{
                "slave": slaveName,
            })
            delete(s.failedSlaves, slaveName)
        }
    }
}

// checkMasterHealth 检查主库健康
func (s *DatabaseFailoverService) checkMasterHealth(ctx context.Context) {
    ticker := time.NewTicker(3 * time.Second)
    defer ticker.Stop()
    
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            // 简单的健康检查查询
            _, err := g.DB("master").GetAll(ctx, "SELECT 1")
            
            s.mu.Lock()
            defer s.mu.Unlock()
            
            if err != nil {
                // 主库不可用
                if s.masterHealthy {
                    g.Log().Warning(ctx, "主库不可用", g.Map{
                        "error": err.Error(),
                    })
                    s.masterHealthy = false
                    
                    // 在生产环境中，这里应该触发主从切换逻辑
                    g.Log().Warning(ctx, "需要手动执行主从切换")
                }
            } else {
                // 主库恢复
                if !s.masterHealthy {
                    g.Log().Info(ctx, "主库恢复")
                    s.masterHealthy = true
                }
            }
        }
    }
}

// GetAvailableSlave 获取可用的从库
func (s *DatabaseFailoverService) GetAvailableSlave() string {
    s.mu.RLock()
    defer s.mu.RUnlock()
    
    availableSlaves := []string{}
    
    if !s.failedSlaves["slave1"] {
        availableSlaves = append(availableSlaves, "slave1")
    }
    
    if !s.failedSlaves["slave2"] {
        availableSlaves = append(availableSlaves, "slave2")
    }
    
    if len(availableSlaves) == 0 {
        return "master" // 如果没有可用的从库，使用主库
    }
    
    // 随机选择一个可用的从库
    return availableSlaves[grand.Intn(len(availableSlaves))]
}

// IsMasterHealthy 主库是否健康
func (s *DatabaseFailoverService) IsMasterHealthy() bool {
    s.mu.RLock()
    defer s.mu.RUnlock()
    
    return s.masterHealthy
}
```

```go
type BackupService struct {

type MysqlBackupConfig struct {

// NewBackupService 创建备份服务实例
func NewBackupService() *BackupService {

        mysqlConfig: &MysqlBackupConfig{

            Port:     g.Cfg().MustGet(ctx, "database.master.port").Int(),
            User:     g.Cfg().MustGet(ctx, "database.master.user").String(),

// StartBackupSchedule 开始备份调度
func (s *BackupService) StartBackupSchedule(ctx context.Context) {
    // 确保备份目录存在
    if err := os.MkdirAll(s.backupDir, 0755); err != nil {
        g.Log().Error(ctx, "创建备份目录失败", g.Map{
            "error": err.Error(),
        })
        return
    }
    
    // 每日凌晨2点执行全量备份
    go s.scheduleFullBackup(ctx)
    
    // 每小时执行增量备份
    go s.scheduleIncrementalBackup(ctx)
}

// scheduleFullBackup 调度全量备份
func (s *BackupService) scheduleFullBackup(ctx context.Context) {
    // 初始延迟到下一个凌晨2点
    now := time.Now()
    nextBackup := time.Date(now.Year(), now.Month(), now.Day(), 2, 0, 0, 0, now.Location())
    if now.After(nextBackup) {
        nextBackup = nextBackup.AddDate(0, 0, 1)
    }
    
    initialDelay := nextBackup.Sub(now)
    time.Sleep(initialDelay)
    
    // 执行备份
    s.performFullBackup(ctx)
    
    // 之后每天凌晨2点执行
    ticker := time.NewTicker(24 * time.Hour)
    defer ticker.Stop()
    
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            s.performFullBackup(ctx)
        }
    }
}

// performFullBackup 执行全量备份
func (s *BackupService) performFullBackup(ctx context.Context) {
    timestamp := time.Now().Format("20060102_150405")
    filename := fmt.Sprintf("%s/full_backup_%s.sql.gz", s.backupDir, timestamp)
    
    // 使用mysqldump执行全量备份
    cmd := exec.Command(
        "mysqldump",
        "-h", s.mysqlConfig.Host,
        "-P", fmt.Sprintf("%d", s.mysqlConfig.Port),
        "-u", s.mysqlConfig.User,
        "-p"+s.mysqlConfig.Password,
        "--single-transaction",
        "--quick",
        "--lock-tables=false",
        s.mysqlConfig.Database,
    )
    
    // 压缩输出
    gzipCmd := exec.Command("gzip")
    
    // 创建管道
    var err error
    cmd.Stdout, err = gzipCmd.StdinPipe()
    if err != nil {
        g.Log().Error(ctx, "创建管道失败", g.Map{
            "error": err.Error(),
        })
        return
    }
    
    // 设置输出文件
    outfile, err := os.Create(filename)
    if err != nil {
        g.Log().Error(ctx, "创建备份文件失败", g.Map{
            "error": err.Error(),
        })
        return
    }
    defer outfile.Close()
    
    gzipCmd.Stdout = outfile
    
    // 启动命令
    if err := gzipCmd.Start(); err != nil {
        g.Log().Error(ctx, "启动gzip命令失败", g.Map{
            "error": err.Error(),
        })
        return
    }
    
    if err := cmd.Start(); err != nil {
        g.Log().Error(ctx, "启动mysqldump命令失败", g.Map{
            "error": err.Error(),
        })
        return
    }
    
    // 等待命令完成
    if err := cmd.Wait(); err != nil {
        g.Log().Error(ctx, "执行mysqldump失败", g.Map{
            "error": err.Error(),
        })
        return
    }
    
    // 关闭输入流
    cmd.Stdout.Close()
    
    if err := gzipCmd.Wait(); err != nil {
        g.Log().Error(ctx, "执行gzip失败", g.Map{
            "error": err.Error(),
        })
        return
    }
    
    g.Log().Info(ctx, "全量备份成功", g.Map{
        "filename": filename,
    })
    
    // 清理过期备份（保留7天）
    s.cleanupOldBackups(ctx, "full_backup_", 7*24*time.Hour)
}

// scheduleIncrementalBackup 调度增量备份
func (s *BackupService) scheduleIncrementalBackup(ctx context.Context) {
    // 每小时执行一次
    ticker := time.NewTicker(1 * time.Hour)
    defer ticker.Stop()
    
    // 立即执行一次
    s.performIncrementalBackup(ctx)
    
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            s.performIncrementalBackup(ctx)
        }
    }
}

// performIncrementalBackup 执行增量备份
func (s *BackupService) performIncrementalBackup(ctx context.Context) {
    timestamp := time.Now().Format("20060102_150405")
    filename := fmt.Sprintf("%s/incremental_backup_%s.sql.gz", s.backupDir, timestamp)
    
    // 增量备份通常使用binlog，这里简化处理
    // 在实际生产环境中，应该使用MySQL的binlog进行增量备份
    
    g.Log().Info(ctx, "增量备份执行", g.Map{
        "filename": filename,
    })
    
    // 清理过期备份（保留24小时）
    s.cleanupOldBackups(ctx, "incremental_backup_", 24*time.Hour)
}

// cleanupOldBackups 清理过期备份
func (s *BackupService) cleanupOldBackups(ctx context.Context, prefix string, maxAge time.Duration) {
    files, err := os.ReadDir(s.backupDir)
    if err != nil {
        g.Log().Error(ctx, "读取备份目录失败", g.Map{
            "error": err.Error(),
        })
        return
    }
    
    now := time.Now()
    for _, file := range files {
        if file.IsDir() {
            continue
        }
        
        if len(file.Name()) > len(prefix) && file.Name()[:len(prefix)] == prefix {
            info, err := file.Info()
            if err != nil {
                continue
            }
            
            if now.Sub(info.ModTime()) > maxAge {
                filepath := fmt.Sprintf("%s/%s", s.backupDir, file.Name())
                if err := os.Remove(filepath); err != nil {
                    g.Log().Error(ctx, "删除过期备份失败", g.Map{
                        "file":  filepath,
                        "error": err.Error(),
                    })
                } else {
                    g.Log().Info(ctx, "删除过期备份成功", g.Map{
                        "file": filepath,
                    })
                }
            }
        }
    }
}
```

package loadbalance

import (
    "hash/crc32"
    "sort"
    "sync"
)

// ConsistentHash 一致性哈希
type ConsistentHash struct {
    mu          sync.RWMutex
    replicas    int               // 虚拟节点数量
    ring        []int             // 哈希环
    nodeMap     map[int]string    // 虚拟节点到真实节点的映射
}

// NewConsistentHash 创建一致性哈希实例
func NewConsistentHash(replicas int) *ConsistentHash {
    return &ConsistentHash{
        replicas: replicas,
        ring:     []int{},
        nodeMap:  make(map[int]string),
    }
}

// AddNode 添加节点
func (c *ConsistentHash) AddNode(node string) {
    c.mu.Lock()
    defer c.mu.Unlock()
    
    // 添加虚拟节点
    for i := 0; i < c.replicas; i++ {
        hash := int(crc32.ChecksumIEEE([]byte(node+string(i))))
        c.ring = append(c.ring, hash)
        c.nodeMap[hash] = node
    }
    
    // 排序哈希环
    sort.Ints(c.ring)
}

// RemoveNode 移除节点
func (c *ConsistentHash) RemoveNode(node string) {
    c.mu.Lock()
    defer c.mu.Unlock()
    
    // 移除虚拟节点
    for i := 0; i < c.replicas; i++ {
        hash := int(crc32.ChecksumIEEE([]byte(node+string(i))))
        delete(c.nodeMap, hash)
        
        // 从环中移除
        for j, v := range c.ring {
            if v == hash {
                c.ring = append(c.ring[:j], c.ring[j+1:]...)
                break
            }
        }
    }
}

// GetNode 获取节点
func (c *ConsistentHash) GetNode(key string) string {
    c.mu.RLock()
    defer c.mu.RUnlock()
    
    if len(c.ring) == 0 {
        return ""
    }
    
    // 计算键的哈希值
    hash := int(crc32.ChecksumIEEE([]byte(key)))
    
    // 二分查找找到第一个大于等于哈希值的节点
    idx := sort.Search(len(c.ring), func(i int) bool {
        return c.ring[i] >= hash
    })
    
    // 如果找不到，使用第一个节点
    if idx == len(c.ring) {
        idx = 0
    }
    
    return c.nodeMap[c.ring[idx]]
}

#### 6.2.1 重试机制设计

对于网络波动等临时性故障，实现智能重试机制：

1. **指数退避重试**：每次重试间隔指数增长，避免立即重试造成的雪崩
2. **最大重试次数限制**：设置最大重试次数，避免无限重试
3. **可重试错误类型判断**：只对临时性错误进行重试
4. **熔断重试**：当重试成功率过低时，触发熔断，停止重试

```go
// 重试工具实现
package retry

import (
    "context"
    "errors"
    "time"
    
    "github.com/gogf/gf/v2/errors/gcode"
    "github.com/gogf/gf/v2/errors/gerror"
    "github.com/gogf/gf/v2/os/glog"
)

// RetryConfig 重试配置
type RetryConfig struct {
    MaxRetries     int           // 最大重试次数
    InitialDelay   time.Duration // 初始延迟
    MaxDelay       time.Duration // 最大延迟
    Factor         float64       // 延迟增长因子
    RetryableError func(error) bool // 判断是否可重试的函数
}

// DefaultRetryConfig 默认重试配置
var DefaultRetryConfig = RetryConfig{
    MaxRetries:   3,
    InitialDelay: 100 * time.Millisecond,
    MaxDelay:     2 * time.Second,
    Factor:       2.0,
    RetryableError: func(err error) bool {
        // 默认认为网络错误、超时错误等可以重试
        code := gerror.Code(err)
        return code == gcode.CodeNil || 
               code == gcode.CodeInternalError ||
               code == gcode.CodeServerBusy
    },
}

// Do 执行重试逻辑
func Do(ctx context.Context, fn func() error, config *RetryConfig) error {
    if config == nil {
        config = &DefaultRetryConfig
    }
    
    var err error
    delay := config.InitialDelay
    
    for attempt := 0; attempt <= config.MaxRetries; attempt++ {
        err = fn()
        if err == nil {
            // 成功，返回
            return nil
        }
        
        // 判断是否需要重试
        if !config.RetryableError(err) {
            // 不可重试的错误，直接返回
            return err
        }
        
        // 最后一次尝试也失败了
        if attempt == config.MaxRetries {
            return gerror.Wrap(err, "达到最大重试次数")
        }
        
        // 记录重试信息
        glog.Warn(ctx, "操作失败，准备重试", 
            glog.Key("attempt", attempt+1),
            glog.Key("max_retries", config.MaxRetries),
            glog.Key("delay", delay),
            glog.Key("error", err.Error()),
        )
        
        // 等待下一次重试
        select {
        case <-time.After(delay):
            // 更新延迟时间，使用指数退避
            delay = time.Duration(float64(delay) * config.Factor)
            if delay > config.MaxDelay {
                delay = config.MaxDelay
            }
        case <-ctx.Done():
            return ctx.Err()
        }
    }
    
    return errors.New("retry failed")
}

    })

        // 处理错误
    }
}
```

```go
    cpuThreshold    float64

    errorThreshold  float64
    rtThreshold     int64
}

// NewDegradeService 创建降级服务实例
func NewDegradeService() *DegradeService {
    return &DegradeService{
        currentLevel:    LevelNormal,
        autoDegrade:     true,
        cpuThreshold:    80.0,
        memoryThreshold: 85.0,
        errorThreshold:  0.05,
        rtThreshold:     500,
    }
}

// SetDegradeLevel 设置降级级别
func (s *DegradeService) SetDegradeLevel(level DegradeLevel) {
    s.mu.Lock()
    defer s.mu.Unlock()
    s.currentLevel = level
}

// GetDegradeLevel 获取当前降级级别
func (s *DegradeService) GetDegradeLevel() DegradeLevel {
    s.mu.RLock()
    defer s.mu.RUnlock()
    return s.currentLevel
}

// IsDegraded 检查是否需要降级特定功能
func (s *DegradeService) IsDegraded(feature string) bool {
    level := s.GetDegradeLevel()
    
    // 根据功能和降级级别判断
    switch feature {
    case "logging":
        return level >= Level1
    case "monitoring":
        return level >= Level1
    case "user-profile":
        return level >= Level2
    case "recommendation":
        return level >= Level2
    case "non-flash-sale":
        return level >= Level3
    case "only-core":
        return level >= Level4
    default:
        return false
    }
}

// StartAutoDegrade 启动自动降级监控
func (s *DegradeService) StartAutoDegrade(ctx context.Context) {
    ticker := time.NewTicker(5 * time.Second)
    defer ticker.Stop()
    
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            if !s.autoDegrade {
                continue
            }
            
            // 检查系统指标
            cpuUsage := getCPUUsage()
            memoryUsage := getMemoryUsage()
            errorRate := getErrorRate()
            avgRT := getAverageRT()
            
            // 根据指标调整降级级别
            newLevel := s.calculateDegradeLevel(cpuUsage, memoryUsage, errorRate, avgRT)
            if newLevel != s.GetDegradeLevel() {
                s.SetDegradeLevel(newLevel)
            }
        }
    }
}

// 计算当前应该的降级级别
func (s *DegradeService) calculateDegradeLevel(cpu, memory, errorRate float64, avgRT int64) DegradeLevel {
    // 简化的降级逻辑，实际应该更复杂
    if cpu > s.cpuThreshold || memory > s.memoryThreshold || 
       errorRate > s.errorThreshold || avgRT > s.rtThreshold {
        
        // 根据严重程度确定降级级别
        if cpu > 90 || memory > 95 || errorRate > 0.2 || avgRT > 1000 {
            return Level4
        } else if cpu > 85 || memory > 90 || errorRate > 0.1 || avgRT > 800 {
            return Level3
        } else if cpu > 80 || memory > 85 || errorRate > 0.05 || avgRT > 500 {
            return Level2
        }
        return Level1
    }
    
    return LevelNormal
}
```

#### 6.2.3 熔断策略设计

结合Sentinel实现多层级的熔断保护，防止系统故障扩散：

1. **熔断触发条件**：
   - 错误率阈值：当错误率超过设定阈值时触发熔断
   - 响应时间阈值：当慢请求比例超过阈值时触发熔断
   - 异常数阈值：当单位时间内异常数超过阈值时触发熔断

2. **熔断状态管理**：
   - 关闭状态：正常处理请求
   - 开启状态：快速失败，不调用后端服务
   - 半开状态：允许少量请求通过，测试服务是否恢复

3. **熔断恢复策略**：
   - 时间窗口恢复：熔断一段时间后自动进入半开状态
   - 成功比例恢复：半开状态下成功请求比例达到阈值后恢复

#### 6.2.4 隔离策略设计

通过资源隔离避免单一服务的故障影响整体系统：

1. **服务隔离**：
   - 秒杀服务独立部署，与常规服务物理隔离
   - 专用的数据库和缓存资源，避免资源竞争
   - 独立的服务实例和线程池，防止相互影响

2. **线程池隔离**：
   - 为不同服务配置独立的线程池
   - 线程池饱和时快速拒绝，避免线程资源耗尽
   - 监控线程池状态，及时发现问题

3. **信号量隔离**：
   - 使用信号量控制并发访问数量
   - 轻量级隔离，适用于低延迟场景
   - 快速失败，避免长时间阻塞

```go
// 线程池隔离实现
package isolation

import (
    "context"
    "sync"
    "time"
    
    "github.com/gogf/gf/v2/container/gpool"
    "github.com/gogf/gf/v2/errors/gcode"
    "github.com/gogf/gf/v2/errors/gerror"
    "github.com/gogf/gf/v2/os/glog"
)

// ThreadPoolIsolation 线程池隔离
type ThreadPoolIsolation struct {
    mu          sync.RWMutex
    threadPools map[string]*gpool.Pool
}

// NewThreadPoolIsolation 创建线程池隔离实例
func NewThreadPoolIsolation() *ThreadPoolIsolation {
    return &ThreadPoolIsolation{
        threadPools: make(map[string]*gpool.Pool),
    }
}

// RegisterThreadPool 注册线程池
func (t *ThreadPoolIsolation) RegisterThreadPool(name string, size int, maxIdle int, idleTimeout time.Duration) {
    t.mu.Lock()
    defer t.mu.Unlock()
    
    pool := gpool.New(size, size, gpool.WithIdleTimeout(idleTimeout))
    t.threadPools[name] = pool
    
    glog.Info(context.Background(), "注册线程池", 
        glog.Key("name", name),
        glog.Key("size", size),
        glog.Key("max_idle", maxIdle),
        glog.Key("idle_timeout", idleTimeout),
    )
}

// ExecuteInPool 在指定线程池中执行任务
func (t *ThreadPoolIsolation) ExecuteInPool(name string, task func() error) error {
    t.mu.RLock()
    pool, exists := t.threadPools[name]
    t.mu.RUnlock()
    
    if !exists {
        return gerror.NewCode(gcode.CodeInvalidOperation, "线程池不存在: "+name)
    }
    
    // 尝试提交任务到线程池
    err := pool.Add(context.Background(), func(ctx context.Context) {
        err := task()
        if err != nil {
            glog.Error(ctx, "任务执行失败", 
                glog.Key("pool", name),
                glog.Key("error", err.Error()),
            )
        }
    })
    
    if err != nil {
        // 线程池已满，拒绝请求
        return gerror.NewCode(gcode.CodeServerBusy, "线程池已满，请稍后再试")
    }
    
    return nil
}

// GetPoolStats 获取线程池状态
func (t *ThreadPoolIsolation) GetPoolStats(name string) (int, int, error) {
    t.mu.RLock()
    pool, exists := t.threadPools[name]
    t.mu.RUnlock()
    
    if !exists {
        return 0, 0, gerror.NewCode(gcode.CodeInvalidOperation, "线程池不存在: "+name)
    }
    
    // 获取线程池状态
    stats := pool.Stats()
    return stats.ActiveCount(), stats.TotalCount(), nil
}

// 初始化服务隔离配置
func InitServiceIsolation() *ThreadPoolIsolation {
    isolation := NewThreadPoolIsolation()
    
    // 注册秒杀服务线程池
    isolation.RegisterThreadPool("flash-sale-pool", 100, 50, 60*time.Second)
    
    // 注册订单服务线程池
    isolation.RegisterThreadPool("order-pool", 50, 20, 60*time.Second)
    
    // 注册库存服务线程池
    isolation.RegisterThreadPool("stock-pool", 80, 30, 60*time.Second)
    
    // 注册支付服务线程池
    isolation.RegisterThreadPool("payment-pool", 30, 10, 60*time.Second)
    
    return isolation
}
```

#### 6.2.5 异步处理与解耦

通过消息队列实现异步处理，提高系统的容错能力：

1. **异步下单流程**：
   - 前端发起秒杀请求
   - 后端快速响应，将请求放入消息队列
   - 消费者服务处理实际的下单逻辑
   - 处理结果通过WebSocket或轮询返回给前端

2. **消息队列选型**：
   - 选择Kafka作为高吞吐量的消息队列
   - 配置合适的分区和副本策略保证可靠性
   - 实现消息幂等性处理，防止重复消费

3. **死信队列处理**：
   - 处理失败的消息进入死信队列
   - 定期分析和处理死信队列中的消息
   - 对于可恢复的错误，实现自动重试机制

```go
// 异步秒杀服务实现
package async

import (
    "context"
    "encoding/json"
    "time"
    
    "github.com/gogf/gf/v2/errors/gcode"
    "github.com/gogf/gf/v2/errors/gerror"
    "github.com/gogf/gf/v2/os/glog"
    "github.com/segmentio/kafka-go"
)

// FlashSaleRequest 秒杀请求
type FlashSaleRequest struct {
    UserID   string `json:"user_id"`
    GoodsID  string `json:"goods_id"`
    Quantity int    `json:"quantity"`
    RequestID string `json:"request_id"`
    Timestamp int64 `json:"timestamp"`
}

// AsyncFlashSaleService 异步秒杀服务
type AsyncFlashSaleService struct {
    producer *kafka.Writer
    consumer *kafka.Reader
    retryConsumer *kafka.Reader
    dlqWriter *kafka.Writer
    orderService *order.OrderService
    stockService *stock.StockService
    redisClient *redis.Client
}

// NewAsyncFlashSaleService 创建异步秒杀服务实例
func NewAsyncFlashSaleService() *AsyncFlashSaleService {
    // 配置Kafka生产者
    producer := kafka.NewWriter(kafka.WriterConfig{
        Brokers: []string{"kafka1:9092", "kafka2:9092", "kafka3:9092"},
        Topic:   "flash-sale-requests",
        Balancer: &kafka.LeastBytes{},
        RequiredAcks: kafka.RequireAll,
        Async: false,
    })
    
    // 配置Kafka消费者
    consumer := kafka.NewReader(kafka.ReaderConfig{
        Brokers: []string{"kafka1:9092", "kafka2:9092", "kafka3:9092"},
        Topic:   "flash-sale-requests",
        GroupID: "flash-sale-consumer-group",
        MinBytes: 10e3,
        MaxBytes: 10e6,
        MaxWait: 1 * time.Second,
    })
    
    // 配置重试队列消费者
    retryConsumer := kafka.NewReader(kafka.ReaderConfig{
        Brokers: []string{"kafka1:9092", "kafka2:9092", "kafka3:9092"},
        Topic:   "flash-sale-retry",
        GroupID: "flash-sale-retry-group",
        MinBytes: 10e3,
        MaxBytes: 10e6,
        MaxWait: 1 * time.Second,
    })
    
    // 配置死信队列生产者
    dlqWriter := kafka.NewWriter(kafka.WriterConfig{
        Brokers: []string{"kafka1:9092", "kafka2:9092", "kafka3:9092"},
        Topic:   "flash-sale-dlq",
        Balancer: &kafka.LeastBytes{},
        RequiredAcks: kafka.RequireAll,
        Async: false,
    })
    
    return &AsyncFlashSaleService{
        producer: producer,
        consumer: consumer,
        retryConsumer: retryConsumer,
        dlqWriter: dlqWriter,
        orderService: order.NewOrderService(),
        stockService: stock.NewStockService(),
        redisClient: redis.NewClient(),
    }
}

// SubmitFlashSaleRequest 提交秒杀请求
func (s *AsyncFlashSaleService) SubmitFlashSaleRequest(ctx context.Context, req *FlashSaleRequest) error {
    // 序列化请求
    data, err := json.Marshal(req)
    if err != nil {
        return gerror.Wrap(err, "序列化请求失败")
    }
    
    // 发送到Kafka
    message := kafka.Message{
        Key:   []byte(req.RequestID),
        Value: data,
        Headers: []kafka.Header{
            {Key: "timestamp", Value: []byte(time.Now().Format(time.RFC3339))},
            {Key: "retry_count", Value: []byte("0")},
        },
    }
    
    err = s.producer.WriteMessages(ctx, message)
    if err != nil {
        return gerror.Wrap(err, "写入消息队列失败")
    }
    
    glog.Info(ctx, "秒杀请求已提交到消息队列", 
        glog.Key("request_id", req.RequestID),
        glog.Key("user_id", req.UserID),
        glog.Key("goods_id", req.GoodsID),
    )
    
    return nil
}

// StartProcessing 开始处理秒杀请求
func (s *AsyncFlashSaleService) StartProcessing(ctx context.Context) {
    // 启动主队列处理
    go s.processMainQueue(ctx)
    
    // 启动重试队列处理
    go s.processRetryQueue(ctx)
    
    // 等待上下文取消
    <-ctx.Done()
    
    // 关闭资源
    s.consumer.Close()
    s.retryConsumer.Close()
    s.producer.Close()
    s.dlqWriter.Close()
}

// 处理主队列
func (s *AsyncFlashSaleService) processMainQueue(ctx context.Context) {
    for {
        select {
        case <-ctx.Done():
            return
        default:
            s.processMessage(ctx, s.consumer)
        }
    }
}

// 处理重试队列
func (s *AsyncFlashSaleService) processRetryQueue(ctx context.Context) {
    for {
        select {
        case <-ctx.Done():
            return
        default:
            s.processMessage(ctx, s.retryConsumer)
        }
    }
}

// 处理消息
func (s *AsyncFlashSaleService) processMessage(ctx context.Context, reader *kafka.Reader) {
    // 读取消息
    m, err := reader.ReadMessage(ctx)
    if err != nil {
        glog.Error(ctx, "读取消息失败", glog.Key("error", err.Error()))
        return
    }
    
    // 解析请求
    var req FlashSaleRequest
    err = json.Unmarshal(m.Value, &req)
    if err != nil {
        glog.Error(ctx, "解析请求失败", glog.Key("error", err.Error()))
        // 处理失败，发送到死信队列
        s.sendToDeadLetter(ctx, m)
        return
    }
    
    // 获取重试次数
    retryCount := 0
    for _, header := range m.Headers {
        if header.Key == "retry_count" {
            retryCount, _ = strconv.Atoi(string(header.Value))
            break
        }
    }
    
    // 处理秒杀逻辑
    err = s.processFlashSale(ctx, &req)
    if err != nil {
        glog.Error(ctx, "处理秒杀请求失败", 
            glog.Key("request_id", req.RequestID),
            glog.Key("retry_count", retryCount),
            glog.Key("error", err.Error()),
        )
        
        // 根据错误类型和重试次数决定是否重试
        if isRetryableError(err) && retryCount < 3 {
            // 发送到重试队列
            s.sendToRetryQueue(ctx, m, retryCount+1)
        } else {
            // 发送到死信队列
            s.sendToDeadLetter(ctx, m)
            
            // 更新订单状态为失败
            s.updateOrderStatus(ctx, req.RequestID, "failed", "")
        }
    } else {
        glog.Info(ctx, "秒杀请求处理成功", 
            glog.Key("request_id", req.RequestID),
            glog.Key("user_id", req.UserID),
            glog.Key("goods_id", req.GoodsID),
        )
    }
}

// 处理秒杀逻辑
func (s *AsyncFlashSaleService) processFlashSale(ctx context.Context, req *FlashSaleRequest) error {
    // 1. 检查重复下单（幂等性保证）
    orderExists, err := s.redisClient.SIsMember(ctx, "flash_sale:orders:"+req.GoodsID, req.UserID).Result()
    if err != nil {
        return gerror.Wrap(err, "检查重复下单失败")
    }
    if orderExists {
        return gerror.NewCode(gcode.CodeInvalidOperation, "用户已抢购过该商品")
    }
    
    // 2. 扣减库存
    success, err := s.stockService.DeductStock(ctx, req.GoodsID, req.Quantity)
    if err != nil {
        return gerror.Wrap(err, "扣减库存失败")
    }
    if !success {
        return gerror.NewCode(gcode.CodeBusinessValidationFailed, "商品库存不足")
    }
    
    // 3. 创建订单
    orderID, err := s.orderService.CreateOrder(ctx, &order.CreateOrderRequest{
        UserID:   req.UserID,
        GoodsID:  req.GoodsID,
        Quantity: req.Quantity,
        Price:    0, // 从商品服务获取价格
        OrderType: "flash_sale",
    })
    if err != nil {
        // 创建订单失败，回滚库存
        s.stockService.RevertStock(ctx, req.GoodsID, req.Quantity)
        return gerror.Wrap(err, "创建订单失败")
    }
    
    // 4. 记录已下单用户，防止重复下单
    err = s.redisClient.SAdd(ctx, "flash_sale:orders:"+req.GoodsID, req.UserID).Err()
    if err != nil {
        glog.Error(ctx, "记录已下单用户失败", glog.Key("error", err.Error()))
        // 这里不返回错误，因为订单已经创建成功
    }
    
    // 5. 更新Redis中的订单状态
    s.updateOrderStatus(ctx, req.RequestID, "success", orderID)
    
    return nil
}

// 判断是否是可重试的错误
func isRetryableError(err error) bool {
    // 网络错误、超时错误等可以重试
    // 业务错误（如库存不足）不重试
    code := gerror.Code(err)
    return code == gcode.CodeNil || 
           code == gcode.CodeInternalError ||
           code == gcode.CodeServerBusy
}

// 发送到重试队列
func (s *AsyncFlashSaleService) sendToRetryQueue(ctx context.Context, m kafka.Message, retryCount int) {
    // 更新重试次数
    headers := make([]kafka.Header, 0, len(m.Headers)+1)
    found := false
    
    for _, header := range m.Headers {
        if header.Key == "retry_count" {
            headers = append(headers, kafka.Header{
                Key:   "retry_count",
                Value: []byte(strconv.Itoa(retryCount)),
            })
            found = true
        } else {
            headers = append(headers, header)
        }
    }
    
    if !found {
        headers = append(headers, kafka.Header{
            Key:   "retry_count",
            Value: []byte(strconv.Itoa(retryCount)),
        })
    }
    
    // 添加重试时间戳
    headers = append(headers, kafka.Header{
        Key:   "retry_timestamp",
        Value: []byte(time.Now().Format(time.RFC3339)),
    })
    
    // 构建重试消息
    retryMessage := kafka.Message{
        Key:     m.Key,
        Value:   m.Value,
        Headers: headers,
    }
    
    // 创建重试队列写入器
    retryWriter := kafka.NewWriter(kafka.WriterConfig{
        Brokers: []string{"kafka1:9092", "kafka2:9092", "kafka3:9092"},
        Topic:   "flash-sale-retry",
        Balancer: &kafka.LeastBytes{},
        RequiredAcks: kafka.RequireAll,
        Async: false,
    })
    
    // 写入重试队列
    err := retryWriter.WriteMessages(ctx, retryMessage)
    if err != nil {
        glog.Error(ctx, "写入重试队列失败", 
            glog.Key("error", err.Error()),
            glog.Key("retry_count", retryCount),
        )
        // 如果写入重试队列失败，直接写入死信队列
        s.sendToDeadLetter(ctx, m)
    } else {
        glog.Info(ctx, "消息已发送到重试队列", 
            glog.Key("retry_count", retryCount),
            glog.Key("key", string(m.Key)),
        )
    }
    
    // 关闭写入器
    retryWriter.Close()
}

// 发送到死信队列
func (s *AsyncFlashSaleService) sendToDeadLetter(ctx context.Context, m kafka.Message) {
    // 添加死信时间戳
    headers := append(m.Headers, kafka.Header{
        Key:   "dlq_timestamp",
        Value: []byte(time.Now().Format(time.RFC3339)),
    })
    
    // 构建死信消息
    dlqMessage := kafka.Message{
        Key:     m.Key,
        Value:   m.Value,
        Headers: headers,
    }
    
    // 写入死信队列
    err := s.dlqWriter.WriteMessages(ctx, dlqMessage)
    if err != nil {
        glog.Error(ctx, "写入死信队列失败", glog.Key("error", err.Error()))
    } else {
        glog.Info(ctx, "消息已发送到死信队列", glog.Key("key", string(m.Key)))
    }
}

// 更新订单状态
func (s *AsyncFlashSaleService) updateOrderStatus(ctx context.Context, requestID, status, orderID string) {
    // 更新Redis中的订单状态
    statusData := map[string]interface{}{
        "status":    status,
        "order_id":  orderID,
        "updated_at": time.Now().Unix(),
    }
    
    err := s.redisClient.HSet(ctx, "flash_sale:order_status:"+requestID, statusData).Err()
    if err != nil {
        glog.Error(ctx, "更新订单状态失败", 
            glog.Key("request_id", requestID),
            glog.Key("error", err.Error()),
        )
    }
    
    // 设置过期时间（7天）
    s.redisClient.Expire(ctx, "flash_sale:order_status:"+requestID, 7*24*time.Hour)
}
```
